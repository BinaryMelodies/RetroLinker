554 static constexpr bool SupportedMagicType(unix_version version, magic_type magic)
555 {
556 switch(version)
557 {
558 default:
559 return false;
560 case Version1:
561 return magic == MAGIC_V1;
562 case Version2:
563 return magic == OMAGIC;
564 case Version3:
565 case Version4:
566 case Version5:
567 return magic == OMAGIC || magic == NMAGIC;
568 case Version6:
569 return magic == OMAGIC || magic == NMAGIC || magic == MAGIC_SEPARATE;
570 case Version7:
571 case SystemIII:
572 case SystemV:
573 return magic == OMAGIC || magic == NMAGIC || magic == MAGIC_SEPARATE || magic == MAGIC_OVERLAY;
574 case AnyBSD:
575 return magic == OMAGIC || magic == NMAGIC
576 /* 16-bit only */
577 || magic == MAGIC_SEPARATE || magic == MAGIC_OVERLAY || magic == MAGIC_AUTO_OVERLAY_NONSEPARATE || magic == MAGIC_AUTO_OVERLAY_SEPARATE
578 /* 32-bit only */
579 || magic == ZMAGIC;
580 case DefaultVersion:
581 return true;
582 }
