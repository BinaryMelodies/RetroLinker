21 {
22 return
23 (uint32_t(uint8_t(type0)) << 24)
24 | (uint32_t(uint8_t(type1)) << 16)
25 | (uint32_t(uint8_t(type2)) << 8)
26 | uint32_t(uint8_t(type3));
27 }
